### 1. 数组创建与属性分析

import numpy as np
arr = np.arange(1,13).reshape(3,4)  # 创建3x4数组
print("形状:", arr.shape)  # (3,4)
print("维度:", arr.ndim)   # 2
print("数据类型:", arr.dtype)  # int32/int64

3. 数组运算与广播
python
# 广播机制示例
A = np.array([[1,2],[3,4],[5,6]])
B = np.array([10,20])
print("广播加法:", A + B)
print("广播乘法:", A * B)
print("点积运算:", np.dot(A, B))
